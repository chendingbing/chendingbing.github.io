使用谷歌浏览器的无界面模式获取HTML进行处理
from selenium import webdriver
import time
from selenium.webdriver.chrome.options import Options
import urllib
import requests
import operator
import xlrd
import xlutils.copy

chrome_options = Options()
chrome_options.add_argument(‘–headless’)
chrome_options.add_argument(‘–disable-gpu’)
chrome_options.binary_location = r’C:\Users\chendingbing\AppData\Local\Google\Chrome\Application\chrome.exe’
driver = webdriver.Chrome(chrome_options=chrome_options)

driver = webdriver.Chrome()

driver.get(“http://www.aiyukemall.com/?page=4&type=reco&sort=0&cate=0")
driver.maximize_window()
js=”var q=document.documentElement.scrollTop=100000”
driver.execute_script(js)
time.sleep(3)
js=”var q=document.documentElement.scrollTop=3600”
driver.execute_script(js)
js=”var q=document.documentElement.scrollTop=1200”
driver.execute_script(js)
js=”var q=document.documentElement.scrollTop=2400”
driver.execute_script(js)
js=”var q=document.documentElement.scrollTop=1800”
driver.execute_script(js)
time.sleep(3)

pageSource = driver.page_source

print(pageSource)

mall_file = xlrd.open_workbook(‘excel/mall.xls’, formatting_info=True)
wt_mall = xlutils.copy.copy(mall_file)
wt_sheet = wt_mall.get_sheet(0)
price = []
prices = driver.find_elements_by_class_name(“pri”)
for x in prices:
if x.text == “”:
pass
else:
price.append(x.text)
for i in range(0,40):

title = driver.find_elements_by_css_selector(".title")[i].text
buy_number = driver.find_elements_by_css_selector(".buy-number")[i].text
image = driver.find_elements_by_class_name("img")[i].find_element_by_class_name("lazy").get_attribute("src")
title = title.split(buy_number)[0]
y = i
i = i+1+120
wt_sheet.write(i, 0, title)
wt_sheet.write(i, 2, buy_number)
wt_sheet.write(i, 3, image)
wt_sheet.write(i, 1, price[y])

wt_mall.save(‘excel/mall.xls’)

print(title)

print(image)

print(buy_number)

wt_mall.save(‘excel/mall.xls’)
driver.quit()
